 public static IRestResponse GetStats() {

 		new HttpBasicAuthenticator("api",
 		                           "key-3ax6xnjp29jd6fds4gc373sgvjxteol0");
 	RestRequest request = new RestRequest();

 	request.AddParameter("event", "opened");
